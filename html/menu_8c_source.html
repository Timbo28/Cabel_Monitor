165/** ***************************************************************************
166 * @brief Display a visualized position to the cable
167 * @param [in] Y-Distance [mm]
168 * @param [in] X-Distance [mm]
169 * @param [in] Current [A]
170 *
171 * Shows the cable position to the device visually and
172 * in mm, when the cable is in range of the device.
173 * When the cable is in a certain range the current will be displayed.
174 * @note Call MENU_visual_init(uint8_t *title) first
175 *****************************************************************************/
176void MENU_visual_act(int16_t x_distance, uint16_t y_distance, float current)
177{
178 char text_position[9]; // in mm
179 char text_current[9]; // in A
180
181 snprintf(text_current, 8, "NaNs A"); // default
182
183 // erase old position
184 BSP_LCD_SetTextColor(LCD_COLOR_WHITE);
185 BSP_LCD_DrawCircle(x_circle_old,y_circle_old+TITLE_HIGHT,10);
186 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,x_circle_old,y_circle_old+TITLE_HIGHT);
187
188 // set static elements
189 BSP_LCD_SetTextColor(LCD_COLOR_BLACK);
190 BSP_LCD_DrawCircle(120,TITLE_HIGHT+220,4); // origin
191 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,10,TITLE_HIGHT+110); // -45°
192 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,120,TITLE_HIGHT+10); // 0°
193 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,230,TITLE_HIGHT+110); // +45°
194
195 BSP_LCD_SetTextColor(LCD_COLOR_LIGHTGRAY); // intermediate steps
196 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,10,TITLE_HIGHT+156); // -60°
197 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,10,TITLE_HIGHT+30); // -30°
198 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,66,TITLE_HIGHT+10); // -15°
199 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,174,TITLE_HIGHT+10); // +15°
200 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,230,TITLE_HIGHT+30); // +30°
201 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,230,TITLE_HIGHT+156); // +60°
202
203 BSP_LCD_SetTextColor(LCD_COLOR_RED);
204
205 // check if cable in range
206 if ( x_distance != CALC_OUTOF_X_RANGE && y_distance != CALC_OUTOF_Y_RANGE ){
207
208 // conversion for display
209 uint16_t x_circle = (uint16_t)(120+x_distance);
210 uint16_t y_circle = 220 - y_distance;
211
212 // display position to device
213 BSP_LCD_DrawCircle(x_circle,y_circle+TITLE_HIGHT,10);
214 BSP_LCD_DrawLine(120,TITLE_HIGHT+220,x_circle,y_circle+TITLE_HIGHT);
215
216 // calculate distance to device
217 snprintf(text_position, 8, "%4d mm",(int)(hypot(x_distance, y_distance)));
218
219 // check if current measurement possible
220 if(current != CURR_OUTOF_Y_RANGE && current != CURR_OUTOF_Angle_RANGE){
221 snprintf(text_current, 9, " %.1f A",(float)(current));
222 }
223
224 // set new position for erase
225 x_circle_old = x_circle;
226 y_circle_old = y_circle;
227 }
228 else{
229 snprintf(text_position, 9, "NaNs mm"); // when no cable detected
230 }
231
232 // display distance to device
233 BSP_LCD_DisplayStringAt(150, TITLE_HIGHT+215, (uint8_t *)text_position,LEFT_MODE);
234 BSP_LCD_DisplayStringAt(20, TITLE_HIGHT+215, (uint8_t *)text_current,LEFT_MODE);
235}

238/** ***************************************************************************
239 * @brief Draw the menu onto the display.
240 *
241 * Each menu entry has two lines.
242 * Text and background colors are applied.
243 * @n These attributes are defined in the variable MENU_draw[].
244 *****************************************************************************/
245void MENU_draw(void)
246{
247 BSP_LCD_SetFont(MENU_FONT);
248 uint32_t x, y, m, w, h;
249 y = MENU_Y;
250 m = MENU_MARGIN;
251 w = BSP_LCD_GetXSize()/MENU_ENTRY_COUNT;
252 h = MENU_HEIGHT;
253 for (uint32_t i = 0; i < MENU_ENTRY_COUNT; i++) {
254 x = i*w;

256 BSP_LCD_FillRect(x+m, y+m, w-2*m, h-2*m);
257 BSP_LCD_SetBackColor(MENU_entry[i].back_color);
258 BSP_LCD_SetTextColor(MENU_entry[i].text_color);
259 BSP_LCD_DisplayStringAt(x+3*m, y+3*m, (uint8_t *)MENU_entry[i].line1, LEFT_MODE);
260 BSP_LCD_DisplayStringAt(x+3*m, y+h/2, (uint8_t *)MENU_entry[i].line2, LEFT_MODE);
261 }
262}

367/** ***************************************************************************
368 * @brief Check for selection/transition
369 *
370 * If the last transition has been consumed (MENU_NONE == MENU_transition)
371 * and the touchscreen has been touched for a defined period
372 * the variable MENU_transition is set to the touched item.
373 * @note Evalboard revision E (blue PCB) has an inverted y-axis
374 * in the touch controller compared to the display.
375 * Uncomment or comment the <b>\#define EVAL_REV_E</b> in main.h accordingly.
376 *****************************************************************************/
377void MENU_check_transition(void)
378{
379 static MENU_item_t item_old = MENU_NONE;
380 static MENU_item_t item_new = MENU_NONE;
381 static TS_StateTypeDef TS_State; // State of the touch controller
382 BSP_TS_GetState(&TS_State); // Get the state
383
384
385// Evalboard revision E (blue) has an inverted y-axis in the touch controller
386#ifdef EVAL_REV_E
387 TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
388#endif
389 // Invert x- and y-axis if LCD ist flipped
390#ifdef FLIPPED_LCD
391 TS_State.X = BSP_LCD_GetXSize() - TS_State.X; // Invert the x-axis
392 TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
393#endif
394
395
396/*
397 #if (defined(EVAL_REV_E) && !defined(FLIPPED_LCD)) || (!defined(EVAL_REV_E) && defined(FLIPPED_LCD))
398 TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
399#endif
400#ifdef EVAL_REV_E
401#endif
402*/
403 if (TS_State.TouchDetected) { // If a touch was detected
404 /* Do only if last transition not pending anymore */
405 if (MENU_NONE == MENU_transition) {
406 item_old = item_new; // Store old item
407 /* If touched within the menu bar? */
408 if ((MENU_Y < TS_State.Y) && (MENU_Y+MENU_HEIGHT > TS_State.Y)) {
409 item_new = TS_State.X // Calculate new item
410 / (BSP_LCD_GetXSize()/MENU_ENTRY_COUNT);
411 if ((0 > item_new) || (MENU_ENTRY_COUNT <= item_new)) {
412 item_new = MENU_NONE; // Out of bounds
413 }
414 if (item_new == item_old) { // 2 times the same menu item
415 item_new = MENU_NONE;
416 MENU_transition = item_old;
417 }
418 } else if((0 < TS_State.Y) && (TITLE_HIGHT > TS_State.Y)){
419 item_new = MENU_CABLE;
420 if (item_new == item_old) { // 2 times the same menu item
421 item_new = MENU_NONE;
422 MENU_transition = item_old;
423 }
424 }
425 else{
426 item_new = MENU_SUBTASK;
427 if (item_new == item_old) { // 2 times the same menu item
428 item_new = MENU_NONE;
429 MENU_transition = item_old;
430 }
431 }
432 }
433 }
434}

438/** ***************************************************************************
439 * @brief Interrupt handler for the touchscreen
440 *
441 * @note BSP_TS_ITConfig(); must be called in the main function
442 * to enable touchscreen interrupt.
443 * @note There are timing issues when interrupt is enabled.
444 * It seems that polling is the better choice with this evaluation board.
445 * @n Call MENU_check_transition() from the while loop in main for polling.
446 *
447 * The touchscreen interrupt is connected to PA15.
448 * @n The interrupt handler for external line 15 to 10 is called.
449 *****************************************************************************/
450void EXTI15_10_IRQHandler(void)
451{
452 if (EXTI->PR & EXTI_PR_PR15) { // Check if interrupt on touchscreen
453 EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
454 if (BSP_TS_ITGetStatus()) { // Get interrupt status
455 BSP_TS_ITClear(); // Clear touchscreen controller int.
456 MENU_check_transition();
457 }
458 EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
459 }
