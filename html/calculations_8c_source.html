1/** ***************************************************************************
2 * @file
3 * @brief Calculates the position to the cable and the current with the values from the ADC(s)
4 *
5 * Cable Position
6 * ===============
7 * To calculate the cable position the function calculate_pos() is using the law of cosines to determine alpha and beta.
8 * With those two angles the function is then calculating the X, Y position and the angle gamma, as shown in the picture below.
9 *
10 * @image html angel_pads.png width=35%
11 * @n a = left pad
12 * @n b = right pad
13 *
14 * @note Special care must be taken with the sign of the X position.
15 * Here the right side of the zero point is negative and the left side is positive.
16 *
17 *
18 * Averaging
19 * =========
20 * By calling calculate_pos() function it is possible to change the number of samples,
21 * which should be averaged before the the distance value get loaded in to the "LPAD_distance" and "RPAD_distance" variables.
22 * With the parameter of this function it is possible to change this number of samples which should be averaged.
23 *
24 * @note The parameter of the calculate_pos function should not be less than 1 and greater than 3,
25 * otherwise the macros FFT_AVG_NUMS must be changed to a higher number.
26 *
27 * Current
28 * =======
29 *
30 * By calling the calculate_pos function the actually current also will be calculated and stored in the "current" variable.
31 * For this, the current can only be calculated if the distance (Y_pos) is between 15 mm and 25 mm and the angle is in the range of -+15 degrees.
32 * If these requirements are not fulfilled, an error code is stored in the "current" variable.
33 *
34 * Error codes
35 * ===========
36 * The whole calculations.c file is using error codes from the header file error_code.h.
37 * This error codes will be stored in the corresponding variables which can not be calculated.
38 *
39 *
40 *
41 *
42 * ----------------------------------------------------------------------------
43 * @author Tim Roos, roostim1@students.zhaw.ch
44 * @date 27.12.2022
45 *****************************************************************************/

161/** ***************************************************************************
162 * @brief Calculate angle, X and Y Position of the cable, from the FFT value.
163 *
164 * @param Number of FFT output values to be averaged before calculating with it.
165 *
166 * @note The zero point is at the leading edge of the device between the two pads.
167 *****************************************************************************/
168void calculate_pos(int fft_avg_num)
169{
170 num_of_samples = fft_avg_num;
171 double cos_alpha;
172 double alpha;
173 double cos_beta;
174 double beta;
175
176 ADC3_IN4_timer_init();
177 ADC3_IN4_timer_start();
178
179 if (MEAS_data_ready){
180 /* Sets the error code as a default value*/
181 X_Pos = CALC_OUTOF_X_RANGE; // ERROR code
182 Y_Pos = CALC_OUTOF_Y_RANGE; // ERROR code
183 Gamma = CALC_OUTOF_ANGLE_RANGE; // ERROR code
184
185 split_Array();
186 calculate_FFT();
187 clear_Buffer();
188 /*Checks if there is no error code from the FFT function*/
189 if(LPAD_FFT_distance != FFT_NO_SIGNAL || RPAD_FFT_distance != FFT_NO_SIGNAL){
190
191 cos_beta = (double)( LPAD_FFT_distance*LPAD_FFT_distance
192 + PAD_SPACING*PAD_SPACING
193 - RPAD_FFT_distance*RPAD_FFT_distance)
194 / (2*LPAD_FFT_distance*PAD_SPACING); // Law of cosines
195
196 cos_alpha = (double)( RPAD_FFT_distance*RPAD_FFT_distance
197 - LPAD_FFT_distance*LPAD_FFT_distance
198 + PAD_SPACING*PAD_SPACING)
199 / (2*RPAD_FFT_distance*PAD_SPACING); // Law of cosines
200 /* Checks if cos_alpha and cos_beta are in range of arccosine*/
201 if(cos_alpha <= 1 && cos_beta <= 1 && cos_alpha > -1 && cos_beta >-1){
202
203 alpha = acos(cos_alpha); //Left angle
204 beta = acos(cos_beta); //Right angle
205
206 if (alpha < M_PI/2 && beta < M_PI/2){// First case : cable in between pads
207 X_Pos = fabs((cos(alpha)*RPAD_FFT_distance - PAD_SPACING/2));
208 Y_Pos = sin(alpha)*RPAD_FFT_distance;
209 }
210 else if (alpha >= M_PI/2 && beta < M_PI/2){ //Second case: cable on the left side of the device
211 X_Pos = cos(M_PI-alpha)*RPAD_FFT_distance + PAD_SPACING/2;
212 Y_Pos = sin(M_PI-alpha)*RPAD_FFT_distance;
213 }
214 else if (alpha < M_PI/2 && beta >= M_PI/2){ //Third case: cable on the right side of the device
215 X_Pos = (cos(M_PI-beta)*LPAD_FFT_distance + PAD_SPACING/2);
216 Y_Pos = sin(M_PI-beta)*LPAD_FFT_distance;
217 }
218
219 Gamma= atan2(Y_Pos,X_Pos );
220 /*sets the correct signs for X_Pos and gamma*/
221 if(alpha < beta){//Check if cable is on the right side of the zero point.
222
223 Gamma = M_PI/2-Gamma;/* Angle minus 90 degree, so that angel is zero degree
224 if cable is in front of the device and a negative angle
225 when we move the cable to the right.*/
226 X_Pos = -X_Pos;
227 }
228 else { // If cable is on the left side of the zero point.
229 Gamma -= M_PI/2; /* 90 degree minus angle, so that angel is zero degree
230 if cable is in front of the device and a positive
231 angle when we move the cable to the left.*/
232 }
233 Gamma=Gamma*RAD_TO_DEGREE;
234
235 calculate_current();
236 check_display_bounderies();
237 }
238 }
239 }
240}
241/** ***************************************************************************
242 * @brief Calculate the current
243 *
244 * with the magnetic field detected by the Hall sensor.
245 *
246 *****************************************************************************/
247void calculate_current(void)
248{
249 /* Checks that the distance between the cable and the Hall sensor is not too large*/
250 if(Y_Pos > 15 && Y_Pos < 25 ){
251
252 /*Checks that the angle between the cable and the Hall sensor is not too large.*/
253 if(Gamma < 15 && Gamma > -15){
254
255 /* Detection of the higher Hall sensor voltage and storage of the higher voltage in the current variable */
256 if(RHALL_FFT_voltage > LHALL_FFT_voltage){
257 current = (RHALL_FFT_voltage*CURRENT_FACTOR*Y_Pos)/1000;
258 }else{
259 current = (LHALL_FFT_voltage*CURRENT_FACTOR*Y_Pos)/1000;
260 }
261
262 }else{
263 current = CURR_OUTOF_Angle_RANGE; // ERROR code
264 }
265 }else{
266 current = CURR_OUTOF_Y_RANGE;// ERROR code
267 }
268}

285/** ***************************************************************************
286 * @brief Transformers the ADC samples in to the frequency domain with a FFT.
287 *
288 * The ADC values at50 Hz from both pads and both Hall will be saved in the
289 * {LPAD_FFT_avg_array[],RPAD_FFT_avg_array[],LHALL_FFT_avg_array[],RHALL_FFT_avg_array[]}
290 * at the position of the avg_counter.
291 *
292 *****************************************************************************/
293void calculate_FFT (void)
294{
295
296 uint8_t RFFT =0; // Flag which declares that an real fast Fourier transformation is required.
297 /* Writes real and imaginary part of different frequencies (from 0 to 320 Hz with delta-frequency) in LPAD_FFT.*/
298 arm_rfft_fast_f32(&fft_handler,LPAD_samples, LPAD_FFT,RFFT);
299 LPAD_FFT_avg_array[avg_counter]=(uint32_t)(hypot(LPAD_FFT[10],LPAD_FFT[11])*sqrt(2)/ADC_NUMS); /* 50 Hz real part is at position 10 and
300 imaginary part is at position 11 of the LPAD_FFT[] array.*/
301 arm_rfft_fast_f32(&fft_handler,RPAD_samples,RPAD_FFT,RFFT);
302 RPAD_FFT_avg_array[avg_counter]= (uint32_t)(hypot(RPAD_FFT[10],RPAD_FFT[11])*sqrt(2)/ADC_NUMS);
303 arm_rfft_fast_f32(&fft_handler,LHALL_samples, LHALL_FFT,RFFT);
304 LHALL_FFT_avg_array[avg_counter]=(uint32_t)(hypot(LHALL_FFT[10],LHALL_FFT[11])*sqrt(2)/ADC_NUMS);
305
306 arm_rfft_fast_f32(&fft_handler,RHALL_samples, RHALL_FFT,RFFT);
307 RHALL_FFT_avg_array[avg_counter]=(uint32_t)(hypot(RHALL_FFT[10],RHALL_FFT[11])*sqrt(2)/ADC_NUMS);
308
309 averaging_FFT_semples();
310}

312/** ***************************************************************************
313 * @brief Averaging several FFT output values for each pad and Hall sensor.
314 *
315 * The number of samples (num_of_samples) which will be averaged, is defined in the function calculate_pos.
316 *
317 * The averages will be saved in the variables {LPAD_FFT_distance, RPAD_FFT_distance, LHALL_FFT_voltage, RHALL_FFT_voltage}.
318 *
319 *****************************************************************************/
320void averaging_FFT_semples(void)
321{
322
323 if(avg_counter == num_of_samples-1){
324
325 LPAD_FFT_distance=0;
326 RPAD_FFT_distance=0;
327 LHALL_FFT_voltage =0;
328 RHALL_FFT_voltage =0;
329
330 for(int i =0; i < num_of_samples; i++){ //If the desired number of samples is achieved, they get summed up.
331
332 LPAD_FFT_distance += LPAD_FFT_avg_array[i];
333 RPAD_FFT_distance += RPAD_FFT_avg_array[i];
334 LHALL_FFT_voltage += LHALL_FFT_avg_array[i];
335 RHALL_FFT_voltage += RHALL_FFT_avg_array[i];
336 }
337
338 LPAD_FFT_distance = LPAD_FFT_distance/(num_of_samples); //The summed up samples get divided by the number of samples to get the average.
339 RPAD_FFT_distance = RPAD_FFT_distance/(num_of_samples);
340 LHALL_FFT_voltage = LHALL_FFT_voltage/(num_of_samples);
341 RHALL_FFT_voltage = RHALL_FFT_voltage/(num_of_samples);
342 avg_counter = 0;
343 distance_LUT();
344 }else{
345 avg_counter++;
346 }
347
348}

406 }
