40/** ***************************************************************************
41 * @brief Initialize Buzzer
42 *
43 * @note Call BUZZER_turn_on() to turn the buzzer on.
44 *****************************************************************************/
45void BUZZER_init(void)
46{
47 __HAL_RCC_GPIOA_CLK_ENABLE(); // Enable clock for port A
48 GPIOA->MODER |= GPIO_MODER_MODER5_0; // Configure PA5 as output
49 GPIOA->BSRR = GPIO_BSRR_BR5; // Bit Reset to turn PA5 off
50
51 __HAL_RCC_TIM5_CLK_ENABLE(); // Enable Clock for TIM5
52 TIM5->PSC = tim_prescaler; // Prescaler for clock freq. = 2kHz
53 TIM5->ARR = TIM_TOP; // Auto reload = counter top value
54 TIM5->CR1 |= TIM_CR1_URS; // Interrupt when overflow/underflow
55 TIM5->DIER |= TIM_DIER_UIE; // Enable Interrupt
56 TIM5->EGR |= TIM_EGR_UG; // Update settings
57
58 NVIC_ClearPendingIRQ(TIM5_IRQn); // Clear pending interrupt on line 0
59 NVIC_EnableIRQ(TIM5_IRQn); // Enable Interrupt
60}

153/** ***************************************************************************
154 * @brief Interrupt when counter of TIM5 is full
155 *
156 * Switches buzzer by default with 2 kHz or with set frequency of BUZZER_set_freq(uint32_t freq)
157 * @note Interrupt will accrue only when BUZZER_get_status(void) = true
158 *****************************************************************************/
159void TIM5_IRQHandler(void)
160{

170}
